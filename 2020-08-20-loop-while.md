# 打印九九乘法表
row = 1
while row <= 9:
    col = 1
    while col <= row:
        print('%d*%d=%d'%(col,row,row*col),end = ' ')
        col = col + 1
        pass
    print() #换行
    row = row + 1
    pass

#打印等腰三角形
row = 1
j = 1

while row <= 5:

    while j <= (2 * row - 1):
        index = 1
        while index <= (9 - 2 * row + 1):
            print(' ', end = "")  #这样输出可以不换行print('.....',end = "")

            index = index + 1
            pass
        # print(index)
        # print()
        # print(j)
        k = 1
        while   k <= j:

            print('*',end = ' ')
            k = k + 1
            pass
        j = j + 2
    print()
    row = row + 1
